### P2P Payment Service

```typescript
// lib/p2p/p2p-payment-service.ts

export class P2PPaymentService {
  private piBlockchain: PiNetworkBlockchain;
  private p2pNetwork: P2PNetwork;
  private walletManager: WalletManager;

  /**
   * Send payment directly to peer
   * Blockchain-verified, P2P settlement
   */
  async sendP2PPayment(
    senderAddress: string,
    recipientAddress: string,
    amount: number
  ): Promise<{
    success: boolean;
    transactionHash: string;
    settlementTime: number;
    blockchainVerified: boolean;
  }> {
    // 1. Verify sender on blockchain
    const sender = await this.piBlockchain.verifyAddress(senderAddress);
    if (!sender) throw new Error('Sender not verified on blockchain');

    // 2. Check sender balance on blockchain
    const balance = await this.piBlockchain.getBalance(senderAddress);
    if (balance < amount) throw new Error('Insufficient balance');

    // 3. Create transaction
    const transaction = await this.piBlockchain.createTransaction({
      from: senderAddress,
      to: recipientAddress,
      amount,
      type: 'p2p_transfer'
    });

    // 4. Sign transaction
    const signedTx = await this.walletManager.signTransaction(transaction);

    // 5. Broadcast to P2P network
    const broadcastResult = await this.p2pNetwork.broadcast(signedTx);

    // 6. Wait for blockchain consensus
    const consensusResult = await this.piBlockchain.waitForConsensus(
      signedTx.hash,
      30000 // 30 second timeout
    );

    // 7. Verify on blockchain
    const verified = await this.piBlockchain.verifyTransaction(
      signedTx.hash
    );

    return {
      success: verified.valid,
      transactionHash: signedTx.hash,
      settlementTime: consensusResult.confirmationTime,
      blockchainVerified: verified.confirmed
    };
  }

  /**
   * Request payment from peer
   * Blockchain-verified request
   */
  async requestP2PPayment(
    requesterAddress: string,
    payerAddress: string,
    amount: number
  ): Promise<{
    success: boolean;
    requestId: string;
    blockchainVerified: boolean;
  }> {
    // Create payment request on blockchain
    const paymentRequest = await this.piBlockchain.createPaymentRequest({
      requester: requesterAddress,
      payer: payerAddress,
      amount,
      expiresAt: Date.now() + 7 * 24 * 60 * 60 * 1000 // 7 days
    });

    // Sign request
    const signedRequest = await this.walletManager.signRequest(paymentRequest);

    // Broadcast to network
    await this.p2pNetwork.broadcast(signedRequest);

    // Verify on blockchain
    const verified = await this.piBlockchain.verifyPaymentRequest(
      signedRequest.id
    );

    return {
      success: verified.valid,
      requestId: signedRequest.id,
      blockchainVerified: verified.confirmed
    };
  }

  /**
   * Discover peers on P2P network
   */
  async discoverPeers(filters?: {
    country?: string;
    minReputation?: number;
    maxLatency?: number;
  }): Promise<Array<{
    address: string;
    reputation: number;
    latency: number;
    isOnline: boolean;
    verifiedOnBlockchain: boolean;
  }>> {
    // Get peers from P2P network
    const peers = await this.p2pNetwork.discoverPeers(filters);

    // Verify each peer on blockchain
    const verifiedPeers = await Promise.all(
      peers.map(async (peer) => {
        const verification = await this.piBlockchain.verifyAddress(
          peer.address
        );
        return {
          ...peer,
          verifiedOnBlockchain: verification?.valid || false
        };
      })
    );

    return verifiedPeers;
  }
}
```

### Blockchain Verification Service

```typescript
// lib/blockchain/pi-blockchain-verification.ts

export class PiBlockchainVerification {
  private horizonClient: HorizonClient;
  private piNetworkClient: PiNetworkClient;

  /**
   * Verify transaction on Pi blockchain
   */
  async verifyTransaction(transactionHash: string): Promise<{
    valid: boolean;
    confirmed: boolean;
    confirmations: number;
    blockNumber: number;
    timestamp: number;
    parties: { sender: string; receiver: string };
    amount: number;
  }> {
    // Query Pi Network for transaction
    const tx = await this.piNetworkClient.getTransaction(transactionHash);

    if (!tx) {
      return {
        valid: false,
        confirmed: false,
        confirmations: 0,
        blockNumber: 0,
        timestamp: 0,
        parties: { sender: '', receiver: '' },
        amount: 0
      };
    }

    // Verify cryptographic signature
    const signatureValid = this.verifySignature(tx);

    // Check confirmation status
    const currentBlock = await this.piNetworkClient.getCurrentBlockNumber();
    const confirmations = currentBlock - tx.blockNumber;
    const confirmed = confirmations >= 6; // 6 confirmations = final

    return {
      valid: signatureValid && tx.valid,
      confirmed,
      confirmations: Math.max(0, confirmations),
      blockNumber: tx.blockNumber,
      timestamp: tx.timestamp,
      parties: {
        sender: tx.from,
        receiver: tx.to
      },
      amount: tx.value
    };
  }

  /**
   * Verify address on blockchain
   */
  async verifyAddress(address: string): Promise<{
    valid: boolean;
    registered: boolean;
    kycVerified: boolean;
    reputation: number;
    totalTransactions: number;
    balance: number;
  }> {
    // Check if address exists on blockchain
    const account = await this.piNetworkClient.getAccount(address);

    if (!account) {
      return {
        valid: false,
        registered: false,
        kycVerified: false,
        reputation: 0,
        totalTransactions: 0,
        balance: 0
      };
    }

    // Verify KYC status on blockchain
    const kycStatus = await this.piNetworkClient.getKYCStatus(address);

    // Get reputation score
    const reputation = await this.piNetworkClient.getReputation(address);

    // Get transaction count
    const txCount = await this.piNetworkClient.getTransactionCount(address);

    // Get balance
    const balance = await this.piNetworkClient.getBalance(address);

    return {
      valid: true,
      registered: true,
      kycVerified: kycStatus.verified,
      reputation: reputation.score,
      totalTransactions: txCount,
      balance
    };
  }

  /**
   * Verify smart contract execution
   */
  async verifySmartContract(contractAddress: string, functionCall: string): Promise<{
    valid: boolean;
    executed: boolean;
    result: unknown;
    gasUsed: number;
    transactionHash: string;
  }> {
    // Get contract from blockchain
    const contract = await this.piNetworkClient.getSmartContract(
      contractAddress
    );

    if (!contract) {
      return {
        valid: false,
        executed: false,
        result: null,
        gasUsed: 0,
        transactionHash: ''
      };
    }

    // Execute contract function
    const result = await this.piNetworkClient.executeSmartContract(
      contractAddress,
      functionCall
    );

    // Verify execution on blockchain
    const verified = await this.piNetworkClient.verifyContractExecution(
      result.transactionHash
    );

    return {
      valid: verified.valid,
      executed: verified.executed,
      result: result.output,
      gasUsed: result.gasUsed,
      transactionHash: result.transactionHash
    };
  }

  /**
   * Verify signature cryptographically
   * @private
   */
  private verifySignature(transaction: any): boolean {
    try {
      // Use Pi Network's cryptographic library
      return this.piNetworkClient.verifySignature(
        transaction.data,
        transaction.signature,
        transaction.publicKey
      );
    } catch (error) {
      console.error('Signature verification failed:', error);
      return false;
    }
  }
}
```
